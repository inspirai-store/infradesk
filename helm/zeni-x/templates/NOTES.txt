  export NODE_PORT=$(kubectl get svc zeni-x -n {{ .Release.Namespace }} -o jsonpath='{.spec.ports[0].nodePort}')

  echo http://$NODE_IP:$NODE_PORT

To get logs, run:

  kubectl logs -n {{ .Release.Namespace }} -l app=zeni-x --tail=100 -f

To port forward to the service, run:

  kubectl port-forward -n {{ .Release.Namespace }} svc/zeni-x 8080:8080 8081:80
